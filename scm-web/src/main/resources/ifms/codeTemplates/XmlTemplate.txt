<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
    "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="${package}.mapper.${entityName}Mapper">

    <select id="select${entityName}List" resultType="java.util.Map">
        SELECT * FROM ${tableName}
        <if test="searchKeyword != null and searchKeyword != ''">
            WHERE COLUMN_NAME LIKE CONCAT('%', #{searchKeyword}, '%')
        </if>
        <!-- 페이징 로직은 데이터베이스별로 다를 수 있음 -->
        LIMIT #{pageNo}, #{pageSize}
    </select>

    <select id="select${entityName}Detail" resultType="java.util.Map">
        SELECT * FROM ${tableName} WHERE ID = #{id}
    </select>

    <insert id="insert${entityName}">
        INSERT INTO ${tableName}(COLUMN1, COLUMN2, ...)
        VALUES(#{field1}, #{field2}, ...)
    </insert>

    <update id="update${entityName}">
        UPDATE ${tableName}
        SET COLUMN1 = #{field1}, COLUMN2 = #{field2}, ...
        WHERE ID = #{id}
    </update>

</mapper>
